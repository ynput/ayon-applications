    @validator("additional_apps")
    def validate_unique_name(cls, value):
        ensure_unique_names(value)
        for item in value:
            if item.name in DEFAULT_APP_GROUPS:
                raise BadRequestException(f"Duplicate name '{item.name}'")
        return value
